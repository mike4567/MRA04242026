
import admin from 'firebase-admin';
import type { App } from 'firebase-admin/app';

let app: App | null = null;

// This function ensures that we only initialize the app once.
function initializeFirebaseAdmin(): App {
  if (app) {
    return app;
  }

  // When running in a Google Cloud environment, the SDK can automatically
  // detect the service account credentials and project ID.
  try {
    if (admin.apps.length === 0) {
        admin.initializeApp({
            credential: admin.credential.applicationDefault(),
        });
        console.log('Firebase Admin SDK initialized successfully.');
    }
    app = admin.app();
    return app;
  } catch (error: any) {
    console.error('Firebase admin initialization error', error);
    // Throwing an error here can help debug initialization issues.
    throw new Error(`Firebase admin initialization failed: ${error.message}`);
  }
}

// These functions ensure that you get an initialized instance of the service.
export const getDb = () => {
  initializeFirebaseAdmin();
  return admin.firestore();
};

export const getStorage = () => {
  initializeFirebaseAdmin();
  return admin.storage();
};
